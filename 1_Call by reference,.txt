<---------------------- CALL BY REFERENCE ------------------------>


#include <iostream>
#include <conio.h>
using namespace std;

void swap(int&x , int&y);
int main()
{
    int a =34, b= 11;
    
    cout << "Before Sawping--------->\n";
    cout << " A = " << a <<endl;
    cout << " B = " << b <<endl;
    
    
    swap(a,b);
    
    cout << "After Sawping--------->\n";
    cout << " A = " << a <<endl;
    cout << " B = " << b <<endl;
    
    getch();
    return 0;
}
void swap(int&x , int&y)
{
    int temp = x;
    x = y;
    y = temp;
}
